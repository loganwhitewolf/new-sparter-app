<design_resolution>
**Ambiguity resolved (RESEARCH.md Pitfall 3 / Open Question 1):**

The detector runs coverage internally using `coveragePatterns` (per D-03: "the detector does the regex matching internally"). It does NOT trust `PatternDetectorRow.covered` for the regex coverage check.

- `PatternDetectorRow.covered: true` is an additional caller-side exclusion hint (used by the caller's dedup logic) — when true, the row is also excluded.
- `PatternDetectorRow.valid: false` is excluded.
- A row matched by ANY entry of `coveragePatterns` (regex test with 'i' flag AND amountSign check) is excluded.

Inclusion predicate: `row.valid === true && row.covered === false && isCovered(row, coveragePatterns) === false`.

**Amount-sign comparison: use Decimal.js** (project rule — `lib/utils/import.ts` already imports it; matches `amountSignMatches` in `categorization.ts`).

**Numeric token check: `/^\d+$/.test(token)`** (integer-only, matches ADR "reference numbers and years").

**Regex escape character set: `/[.*+?^${}()|[\]\\]/g`** (standard MDN set, covers all metacharacters the ADR mentions).

**Sample descriptions: drawn from `row.description` (raw), NOT `row.normalizedDescription`** (RESEARCH.md Pitfall 4).

**Grouping algorithm (RESEARCH.md Pitfall 1 — must group by prefix, not exact sequence):**
For each pair of qualifying rows (post-strip token list with length >= 2), compute their longest common token prefix. If LCP length >= 2, they belong to a common group. Concretely: bucket rows by their first stripped token; within each bucket, compute the longest prefix shared by ALL members (intersect down). If the bucket has >= 2 rows and the shared prefix has >= 2 tokens, emit one suggestion. This guarantees: (a) "pagamento pos 12345" + "pagamento pos supermercato" share the 2-token prefix "pagamento pos"; (b) "pagamento pos negozio rossi" + "pagamento pos negozio bianchi" + "pagamento pos negozio verdi" preserve the 3-token prefix "pagamento pos negozio" (longest qualifying), not truncated to 2.

**Edge case: all amounts null in a group → `detectedAmountSign: 'any'`** (RESEARCH.md Pitfall 5).

**Detector does NOT sort or slice to 5.** The 5-suggestion cap (ANL-03) is phase 34 scope. Detector returns all qualifying suggestions in deterministic order (e.g. insertion order from grouping pass).
</design_resolution>

    **Module structure after this task:**
    ```typescript
    import Decimal from 'decimal.js'

    // (the three exported interfaces from Task 1 — unchanged)

    // --- Internal helpers (unexported, lowercase camelCase) ---

    function isNumericToken(token: string): boolean {
      return /^\d+$/.test(token)
    }

    function escapeRegex(str: string): string {
      return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')
    }

    function amountSignMatches(
      amountSign: 'positive' | 'negative' | 'any',
      amount: string | null,
    ): boolean {
      if (amountSign === 'any') return true
      if (amount === null) return true
      try {
        const d = new Decimal(amount)
        if (amountSign === 'positive') return d.greaterThanOrEqualTo(0)
        if (amountSign === 'negative') return d.lessThan(0)
      } catch {
        // unparseable amount — treat as match (defensive: do not lose row to bad data)
      }
      return true
    }

    function isCoveredByPatterns(
      row: PatternDetectorRow,
      coveragePatterns: CoveragePattern[],
    ): boolean {
      for (const p of coveragePatterns) {
        try {
          const regex = new RegExp(p.pattern, 'i')
          if (regex.test(row.normalizedDescription) && amountSignMatches(p.amountSign, row.amount)) {
            return true
          }
        } catch {
          // invalid regex pattern — skip and continue
        }
      }
      return false
    }

    function stripNumericTokens(normalized: string): string[] {
      return normalized.split(/\s+/).filter(t => t.length > 0 && !isNumericToken(t))
    }

    function longestCommonPrefix(a: string[], b: string[]): string[] {
      const result: string[] = []
      const len = Math.min(a.length, b.length)
      for (let i = 0; i < len; i++) {
        if (a[i] !== b[i]) break
        result.push(a[i])
      }
      return result
    }

    function inferAmountSign(amounts: (string | null)[]): 'positive' | 'negative' | 'any' {
      const signs = new Set<'positive' | 'negative'>()
      for (const amount of amounts) {
        if (amount === null) continue
        try {
          const d = new Decimal(amount)
          if (d.lessThan(0)) signs.add('negative')
          else signs.add('positive')
        } catch {
          // unparseable — skip
        }
      }
      if (signs.size === 1) {
        return signs.has('positive') ? 'positive' : 'negative'
      }
      return 'any'
    }

    // --- Public function ---

    export function detectPatternSuggestions(
      rows: PatternDetectorRow[],
      coveragePatterns: CoveragePattern[],
    ): PatternSuggestion[] {
      // Step 1: filter eligible rows (valid, not caller-flagged covered, not coverage-pattern-matched)
      // and compute stripped tokens; drop rows with <2 stripped tokens.
      type Candidate = { row: PatternDetectorRow; tokens: string[] }
      const candidates: Candidate[] = []
      for (const row of rows) {
        if (!row.valid) continue
        if (row.covered) continue
        if (isCoveredByPatterns(row, coveragePatterns)) continue
        const tokens = stripNumericTokens(row.normalizedDescription)
        if (tokens.length < 2) continue
        candidates.push({ row, tokens })
      }

      // Step 2: bucket by first stripped token (any group sharing a >=2-token prefix
      // must share the first token by definition).
      const buckets = new Map<string, Candidate[]>()
      for (const c of candidates) {
        const head = c.tokens[0]
        const list = buckets.get(head) ?? []
        list.push(c)
        buckets.set(head, list)
      }

      // Step 3: for each bucket with >=2 members, compute the longest prefix
      // shared by ALL members (intersect down). If the shared prefix has >=2 tokens,
      // emit one suggestion.
      const suggestions: PatternSuggestion[] = []
      for (const group of buckets.values()) {
        if (group.length < 2) continue
        let prefix = group[0].tokens
        for (let i = 1; i < group.length; i++) {
          prefix = longestCommonPrefix(prefix, group[i].tokens)
          if (prefix.length < 2) break
        }
        if (prefix.length < 2) continue

        const prefixString = prefix.join(' ')
        const escaped = escapeRegex(prefixString)
        const amounts = group.map(g => g.row.amount)
        const sampleDescriptions = group.slice(0, 3).map(g => g.row.description)

        suggestions.push({
          pattern: escaped,
          matchCount: group.length,
          detectedAmountSign: inferAmountSign(amounts),
          sampleDescriptions,
        })
      }

    }
    ```
