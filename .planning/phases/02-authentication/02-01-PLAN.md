import { drizzle } from 'drizzle-orm/mysql2'

const pool = mysql.createPool({

**File 1: auth.ts (project root)**

```typescript
import { betterAuth } from 'better-auth'
import { drizzleAdapter } from 'better-auth/adapters/drizzle'
import { nextCookies } from 'better-auth/next-js'
import { db } from '@/lib/db'

export const auth = betterAuth({
  secret: process.env.BETTER_AUTH_SECRET!,
  baseURL: process.env.BETTER_AUTH_URL,
  emailAndPassword: {
    enabled: true,
    minPasswordLength: 8,
    autoSignIn: true,
  },
  database: drizzleAdapter(db, {
    provider: 'mysql',
  }),
  user: {
    additionalFields: {
      subscriptionPlan: {
        type: ['free', 'basic', 'pro'] as const,
        required: false,
        defaultValue: 'free',
        input: false,
      },
      role: {
        type: ['user', 'admin'] as const,
        required: false,
        defaultValue: 'user',
        input: false,
      },
    },
  },
  plugins: [nextCookies()],
})
```

Constraints to verify:
- `plugins: [nextCookies()]` MUST be the last (and only) plugin in the array. If placed before others it breaks cookie propagation in Server Actions.
- `autoSignIn: true` implements D-02 (post-registration auto-login → redirect to /dashboard).
- `input: false` on both additionalFields implements D-09 (users cannot set these at signup).
- `minPasswordLength: 8` implements D-11.
- `baseURL: process.env.BETTER_AUTH_URL` (not hardcoded URL).
