```typescript
import { z } from 'zod'
import { DASHBOARD_PRESETS } from '@/lib/utils/date'

export const DashboardPresetSchema = z.enum(DASHBOARD_PRESETS).default('last-month')
export const DashboardTypeSchema = z.enum(['out', 'in', 'all']).default('out')

export const DashboardFiltersSchema = z.object({
  preset: DashboardPresetSchema,
  type: DashboardTypeSchema,
})

export type DashboardPreset = z.infer<typeof DashboardPresetSchema>
export type DashboardType = z.infer<typeof DashboardTypeSchema>
export type DashboardFilters = z.infer<typeof DashboardFiltersSchema>

export function parseDashboardFilters(input: {
  preset?: string | string[]
  type?: string | string[]
}): DashboardFilters {
  const rawPreset = Array.isArray(input.preset) ? input.preset[0] : input.preset
  const rawType = Array.isArray(input.type) ? input.type[0] : input.type
  return DashboardFiltersSchema.parse({
    preset: rawPreset ?? 'last-month',
    type: rawType ?? 'out',
  })
}
```

Export `getOverview = cache(async (): Promise<OverviewData> => { ... })`.

Implementation rules:
- Call `const { userId } = await verifySession()`.
- Compute current month range from `new Date()` and previous month range explicitly.
- Query current and previous `uncategorizedCount` with `expense.status = '1'`, `expense.userId = userId`, date range, and category join excluding `category.slug = 'ignore'`.
- Because amounts do not exist yet, return `totalIn: '0.00'`, `totalOut: '0.00'`, `balance: '0.00'`, `savingsRate: 0`.
- Deltas for monetary metrics are `null`; `uncategorizedCount` delta uses `computeDeltaPercent(current, previous)`.

Export `getCategoriesBreakdown = cache(async (filters: DashboardFilters): Promise<BreakdownCategory[]> => { ... })`.

Implementation rules:
- Call `verifySession()`.
- Use `dashboardPresetToDateRange(filters.preset)`.
- Join `expense -> subCategory -> category`.
- Conditions always include `eq(expense.userId, userId)`, `gte(expense.createdAt, from)`, `lte(expense.createdAt, to)`, `isNotNull(expense.subCategoryId)`, and `ne(category.slug, 'ignore')`.
- If `filters.type === 'out'`, add `eq(category.type, 'out')`.
- If `filters.type === 'in'`, add `eq(category.type, 'in')`.
- If `filters.type === 'all'`, add `inArray(category.type, ['in', 'out'])`.
- Group rows in JS into category objects with nested subcategory arrays.
- Set every `amount` to `'0.00'`.
- Use `computeBreakdownPercentages()` to calculate count-based `percentage` for categories and subcategories.
- Sort categories by `count` descending and return at most 15 categories.

Use Drizzle `count()` from `drizzle-orm` for counts. If TypeScript needs numeric conversion, use `Number(row.count ?? 0)` only for counts, never for monetary amounts.
