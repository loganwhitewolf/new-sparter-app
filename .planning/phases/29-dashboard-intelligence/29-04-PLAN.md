      function sortKey(item: CategoryRankingItem, deviations?: Map<number, DeviationData>): number {
        // Numeric bucket: 0 (top). 'new' bucket: 1. null/missing bucket: 2.
        if (!deviations) return 3
        const entry = deviations.get(item.id)
        if (entry === undefined || entry.deviation === null) return entry?.isNew ? 1 : 2
        return 0
      }

      function compareForSort(
        a: CategoryRankingItem,
        b: CategoryRankingItem,
        sort: DashboardSort,
        deviations?: Map<number, DeviationData>
      ): number {
        if (sort !== 'deviation' || !deviations) return 0  // preserve DAL order (amount desc)
        const ka = sortKey(a, deviations)
        const kb = sortKey(b, deviations)
        if (ka !== kb) return ka - kb
        if (ka === 0) {
          // both numeric — compare absolute deviation desc
          const da = Math.abs(deviations.get(a.id)!.deviation as number)
          const db = Math.abs(deviations.get(b.id)!.deviation as number)
          if (da !== db) return db - da
        }
        // tiebreak: amount desc
        return Number(b.amount) - Number(a.amount)
      }
